from math import sqrt

class Rectangle:
    def __init__(self,width:int,height:int):
        self.width=width
        self.height=height
    
    def set_width(self,new_widht):
        self.width = new_widht
    
    def set_height(self,new_height):
        self.height = new_height
    
    def get_area(self)-> float:
        return self.width * self.height
    
    def get_perimeter(self)-> float:
        return 2* (self.width +  self.height )
    
    def get_diagonal(self)-> float:
        return sqrt(self.width ** 2 + self.height ** 2)
        
    
    def get_picture(self) -> str:
        if (self.width > 50 or self.height > 50):
            return "Too big for picture."
        
        linea = "*" * int(self.width) + "\n"
        return linea * int(self.height)
    
    def get_amount_inside(self, shape) -> int:
       
        horizontal_fit = self.width // shape.width
        vertical_fit = self.height // shape.height
        
        return int(horizontal_fit * vertical_fit)
    
    def __str__(self):
        return (f"Rectangle(width={self.width}, height={self.height})")

class Square(Rectangle):
    def __init__(self,side:int):
        super().__init__(side,side)
    
    def set_side(self,side:int):
        self.width=side
        self.height = side
    
    def set_width(self, width):
        """Sobrescribe el método de Rectangle para mantener el cuadrado."""
        self.set_side(width)

    def set_height(self, height):
        """Sobrescribe el método de Rectangle para mantener el cuadrado."""
        self.set_side(height)
    
    def __str__(self)-> str:
        return f"Square(side={self.width})"
    
    
        


rect = Rectangle(3, 6)
print(rect)  # Salida: Rectangle(width=3, height=6)

# 2. Cambiamos sus valores
rect.set_width(10)
rect.set_height(20)

# 3. La representación cambia automáticamente
print(rect)  # Salida: Rectangle(width=10, height=20)
