def quick_sort(array):
    if not array:
        return []
    
    pivote = array[-1]
    less_pivote = [x for x in  array if x < pivote]
    equal_pivote= [x for x in  array if x == pivote]
    mayor_pivote =[x for x in  array if x > pivote]
   
    return quick_sort(less_pivote) + equal_pivote + quick_sort(mayor_pivote)
