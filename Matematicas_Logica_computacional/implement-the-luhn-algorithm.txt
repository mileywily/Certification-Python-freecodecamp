def verify_card_number(card_number):
    # 1. Eliminar guiones y espacios
    card_number = card_number.replace("-", "").replace(" ", "")
    
    # Invertimos la cadena para trabajar de derecha a izquierda más fácilmente
    # El dígito de verificación (el último) queda en la posición 0
    digits = [int(d) for d in card_number[::-1]]
    
    # 2. Procesar los dígitos (duplicar cada segundo dígito)
    # El dígito de verificación (índice 0) NO se duplica.
    # Los dígitos en posiciones impares (1, 3, 5...) se duplican.
    for i in range(1, len(digits), 2):
        digits[i] *= 2
        
        # 3. Si el resultado es mayor a 9, sumamos sus dígitos o restamos 9
        if digits[i] > 9:
            digits[i] -= 9
            
    # 4. Sumar todos los números
    total_sum = sum(digits)
    
    # 5. Verificar si es múltiplo de 10
    if total_sum % 10 == 0:
        return "VALID!"
    else:
        return "INVALID!"
