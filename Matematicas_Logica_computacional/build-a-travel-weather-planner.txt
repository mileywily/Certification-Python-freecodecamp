distance_mi =100
is_raining =False
has_bike =True
has_car =True
has_ride_share_app = True

if distance_mi == False:
    print(False)
if distance_mi <= 1:
    if not is_raining:
      print(True)
    else:
      print(False)
elif  distance_mi <= 6:
    if not is_raining and not has_bike:
        print(False)
    elif is_raining and not has_bike:
        print(False)
    else:
        print(True)
else:
    if distance_mi > 6 and not has_car and not has_ride_share_app:
        print(False)
    else:
        print(True)
